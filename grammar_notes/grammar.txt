smts -> (cmd|cmt)*
cmd -> variable|fun

variable -> (new|old)';'
new -> (CONST)? TYPE ID '=' (STRING|NUM)
old -> ID '=' (STRING|NUM)


fun -> def|(call ';')
call -> CALLID '(' params ')'
params -> param | paramc+ param | /*empty*/
param -> STRING|NUM
paramc -> param ',' | param ',' paramc

def -> TYPE defid '(' params ')' '{' smts '}'
defid -> SETUP | LOOP | ID
